          // 期末考试 ：删除min< x < max 的值 
          struct ListNode{
            int data;
            ListNode* next;
            ListNode(x) : data(x),next(NULL){}
          };
          ListNode* yes_no(ListNode* head,int min_x,max_x)
          {
            ListNode* p = new ListNode(0);
            p->next = head;
            ListNode* m = p;
            ListNode* n = head;
            
            while(n != NULL)
            {
              if(n->data > min_x && n->data < max_x){
                m->next = n->next;
                delete n;
                n = m->next;
              }
              else{
                m = n;
                n = n->next;
              }
            }
            ListNode* newHead = p->next;
            delete p;
            return newHead;
          }
